HOW TO RUN PARSE_NODE (from repo root)

  backend/.venv/bin/python - <<'PY'
  import sys; sys.path.insert(0, '.')
  import sqlite3
  from datetime import datetime
  from backend.crawler.models import CrawlRecord
  from backend.agents.nodes.parse_node import parse_node

  # Load a product-page record from DB (or build one manually)
  conn = sqlite3.connect('data/crawl.db')
  rows = conn.execute('SELECT url, content_type, text, fetched_at FROM crawl_records LIMIT 5').fetchall()
  conn.close()

  records = [CrawlRecord(url=r[0], fetched_at=datetime.fromisoformat(r[3]),
             status_code=200, content_type=r[1], content_hash='', text=r[2] or '') for r in rows]

  state = {'urls_to_crawl': [], 'crawl_records': records,
           'products': [], 'errors': [], 'metrics': {'crawl_count': len(records)}}
  state = parse_node(state)
  print(f"Parsed {len(state['products'])} products")
  PY
